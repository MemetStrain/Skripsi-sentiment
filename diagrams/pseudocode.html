Algoritma 1 — Pipeline Update Harian

Sumber: scheduler/main.py::run_daily_update. Lima langkah berurutan dengan dedup berbasis CSV lokal sebagai single source of truth. 

ALGORITHM run_daily_pipeline
SOURCE: scheduler/main.py :: run_daily_update

INPUT:
    paths_dict    // {cpo_raw, cpo_preproc, news_raw, news_preproc, news_sent}
    db            // Firestore client (init_firebase)

OUTPUT:
    side effects on local CSV + Firestore mirrors; log lines

BEGIN
    // ── Langkah 1: PRICE ───────────────────────────────────────
    cutoff   = most_recent_trading_day()
    have     = latest_price_date(paths.cpo_raw)
    IF have IS NULL OR have < cutoff THEN
        price = fetch_latest_price()       // investiny, lookback 90 / 365 / 1095 d
        IF price IS NOT NULL AND price.date > have THEN
            append_price_row_indonesian(paths.cpo_raw, price)
            preprocess_price_csv(paths.cpo_raw, paths.cpo_preproc)
            write_price(db, price)         // daily_prices/{date}
        END IF
    END IF

    // ── Langkah 2: NEWS (scrape + preprocess + sentiment) ──────
    have_news = latest_news_date(paths.news_sent)
    new_articles = []
    news_changed = FALSE
    IF have_news IS NULL OR have_news < cutoff THEN
        raw = scrape_new_articles(have_news)            // MPOB, dedup by URL
        IF raw IS NOT EMPTY THEN
            append_news_rows(paths.news_raw, raw, RAW_FIELDS)
            cleaned = preprocess_articles(raw)          // NewsPreprocessor
            IF cleaned IS NOT EMPTY THEN
                append_news_rows(paths.news_preproc, cleaned, RAW_FIELDS)
                scored = run_sentiment_on_articles(cleaned)  // FinBERT-Tone
                append_news_rows(paths.news_sent, scored, SENTIMENT_FIELDS)
                new_articles = scored
                news_changed = TRUE
            END IF
        END IF
    END IF

    // ── Langkah 3: mirror artikel ke Firestore ─────────────────
    IF news_changed AND new_articles IS NOT EMPTY THEN
        write_news_articles(db, new_articles)           // doc ID = md5(url)
    END IF

    // ── Langkah 4: sentiment_aggregates ────────────────────────
    IF news_changed AND new_articles IS NOT EMPTY THEN
        affected_dates = { a.date FOR a IN new_articles }
        all_articles   = load_news_from_csv(paths.news_raw, paths.news_sent)
        affected       = [ a FOR a IN all_articles
                            IF a.date IN affected_dates ]
        aggregates = compute_sentiment_aggregates(affected)
        write_sentiment_aggregates(db, aggregates)
    ELSE IF _is_aggregates_empty(db) THEN
        all_articles = load_news_from_csv(paths.news_raw, paths.news_sent)
        aggregates   = compute_sentiment_aggregates(all_articles)
        write_sentiment_aggregates(db, aggregates)
    END IF

    // ── Langkah 5: HMM state decoding (frozen params) ──────────
    update_hmm_states(db)                              // see Algorithm 2 (decode)

    RETURN log_DAILY_UPDATE_COMPLETE
END

Algoritma 2 — Training HMM dengan K-Means Multi-Restart (Offline)

Sumber: markov/cpo_hmm_states.py. Dilakukan offline (tidak per-hari). Outputnya adalah parameter beku yang dipublikasikan ke hmm_models/Daily oleh migrate_hmm_to_firestore.py; scheduler harian hanya mendecode menggunakan parameter ini (online forward filter, bukan re-fit). 

ALGORITHM fit_hmm_with_restarts
SOURCE: markov/cpo_hmm_states.py :: fit_hmm_with_restarts

INPUT:
    X            // (T, 5) matriks fitur: Log_Return_Z, Volatility_Z,
                 //                       RSI_norm, MACD_Z, BB_Width_Z
    n_states     = 3              // Bearish / Neutral / Bullish (manual lock)
    cov_type     = "full"
    n_iter       = 1000           // max EM iter per restart
    tol          = 1e-4
    n_restarts   = 50
    base_seed    = 42

OUTPUT:
    best_model       // GaussianHMM dengan log-likelihood tertinggi
    best_score       // nilai log-likelihood tersebut

BEGIN
    best_model = NULL
    best_score = -INFINITY

    FOR i = 0 TO n_restarts - 1 DO
        seed = base_seed + i
        (model, score) = _fit_single(X, n_states, cov_type, n_iter, tol, seed)
        IF score > best_score THEN
            best_score = score
            best_model = model
        END IF
    END FOR

    RETURN (best_model, best_score)
END


ALGORITHM _fit_single
SOURCE: markov/cpo_hmm_states.py :: _fit_single

INPUT:
    X, n_states, cov_type, n_iter, tol, seed

OUTPUT:
    (model, log_likelihood)  // atau (NULL, -INFINITY) saat error

BEGIN
    TRY
        rng = RandomState(seed)
        D   = X.shape[1]                  // = 5

        // (1) K-Means warm start untuk means & covars
        km          = KMeans(n_clusters = n_states, random_state = seed, n_init = 10)
        km.fit(X)
        init_means  = km.cluster_centers_
        labels      = km.labels_
        init_covars = []
        FOR s = 0 TO n_states - 1 DO
            cluster_X = X[labels == s]
            cov       = cov(cluster_X.T)  IF |cluster_X| > 1 ELSE I(D)
            cov       = cov + I(D) * 1e-3 // diagonal floor (numerical stability)
            init_covars.append(cov)
        END FOR

        // (2) Transition matrix: uniform + small noise → renormalised
        transmat = full((n_states, n_states), 1/n_states)
        transmat = transmat + rng.uniform(-0.05, 0.05, transmat.shape)
        transmat = |transmat|
        transmat = transmat / row_sum(transmat)

        // (3) Start probabilities: uniform + small noise → renormalised
        startprob = full(n_states, 1/n_states)
        startprob = startprob + rng.uniform(-0.05, 0.05, n_states)
        startprob = |startprob|
        startprob = startprob / sum(startprob)

        // (4) Build hmmlearn GaussianHMM with custom init
        model = GaussianHMM(
            n_components    = n_states,
            covariance_type = cov_type,
            n_iter          = n_iter,
            tol             = tol,
            random_state    = seed,
            init_params     = "",         // jangan timpa nilai seeded
            params          = "stmc",     // optimasi startprob/transmat/means/covars
            min_covar       = 1e-3)
        model.startprob_ = startprob
        model.transmat_  = transmat
        model.means_     = init_means
        model.covars_    = init_covars    // (sesuaikan bentuk per cov_type)

        // (5) Baum-Welch EM
        model.fit(X)
        RETURN (model, model.score(X))

    CATCH exception
        RETURN (NULL, -INFINITY)
    END TRY
END

Algoritma 3 — Inferensi Forecast (Precompute oleh Scheduler)

Sumber: prediction/inference.py::compute_forecast_trails (dipanggil dari scheduler/precompute_forecasts.py::precompute_and_write sebagai langkah 5 run_daily_update / run_initial_load). Untuk setiap horizon h ∈ {1..7}, model XGBoost pemenang (lowest base MAPE per horizon) dipakai untuk memprediksi log-return, lalu dieksponensiasi menjadi harga prediksi. Hasil ditulis ke koleksi forecasts (per-titik, scalar fields) + dokumen forecast_meta/Daily (winners + metrics dibundel ke payload_json). 

ALGORITHM compute_forecast_trails
SOURCE: prediction/inference.py :: compute_forecast_trails

INPUT:
    db              // Firestore client
    max_horizon  = 7         // clamped to [1, 7]
    window_days  = 90        // clamped to [7, 365]

OUTPUT:
    payload (dict): {
        horizons, winners, configs, trails[], metrics, generated_at }

BEGIN
    // (1) winners.json menentukan tag pemenang per horizon
    payload  = load_winners()                  // lru_cache(maxsize=1)
    winners  = payload.winners_by_horizon      // {h: tag in {cpo_only,cpo_hmm,cpo_sentiment,full}}
    configs  = payload.configs_by_horizon      // {h: "C1"|"C2"|"C3"|"C4"}

    // (2) Bangun frame fitur penuh dari Firestore
    df = build_inference_frame(db)
    IF df IS EMPTY THEN
        RETURN empty_payload()
    END IF

    dates_sorted = df.Date.tolist()
    last_idx     = |dates_sorted| - 1
    next_td      = _next_trading_day_after(dates_sorted[last_idx])  // skip weekends

    // (3) Indeks anchor dalam window 90 hari terakhir
    cutoff     = df.Date.max() - timedelta(days = window_days)
    window_idx = indices where df.Date >= cutoff

    trails = []
    FOR h = 1 TO max_horizon DO
        tag = winners.get(h)
        IF tag IS NULL THEN CONTINUE

        TRY
            (model, scaler, feature_cols) = load_model(tag, h, "csa")  // lru_cache 64
        CATCH FileNotFoundError
            log_warning("missing model for h=" + h)
            CONTINUE
        END TRY

        points = []
        FOR i IN window_idx DO
            target_idx = i + h
            IF target_idx > last_idx + 1 THEN CONTINUE   // di luar future point
            pred_date = dates_sorted[target_idx]   IF target_idx ≤ last_idx
                        ELSE next_td

            row = df.iloc[[i]]
            TRY
                X = select_feature_matrix(row, feature_cols)
            CATCH ValueError
                CONTINUE   // fitur tidak lengkap pada anchor i
            END TRY

            IF any_nan(X) THEN CONTINUE
            IF scaler IS NOT NULL THEN X = scaler.transform(X)

            log_ret      = model.predict(X)[0]                          // XGBoost regression
            anchor_close = row.Close.iloc[0]
            pred_price   = anchor_close * exp(clip(log_ret, -10, 10))

            points.append({
                anchor_date     : dates_sorted[i],
                anchor_price    : round(anchor_close, 2),
                predicted_date  : pred_date,
                predicted_price : round(pred_price, 2),
                log_return      : round(log_ret, 6) })
        END FOR

        trails.append({ horizon: h, tag: tag, config: configs[h], points: points })
    END FOR

    RETURN {
        horizons     : 1..max_horizon,
        winners      : winners,
        configs      : configs,
        trails       : trails,
        metrics      : payload.metrics,
        tag_to_config: payload.tag_to_config,
        generated_at : utcnow_isoformat() + "Z" }
END


PROCEDURE build_inference_frame
SOURCE: prediction/inference.py :: build_inference_frame

BEGIN
    price_df      = _load_prices(db)              // daily_prices
    sent_df       = _load_sentiment(db)           // sentiment_aggregates Daily
    hmm_state_df  = _load_hmm_state(db)           // hmm_states Daily

    df = add_price_features(price_df)             // OHLCV-derived
    df = add_hmm_derived_features(df)             // RSI/MACD/Bollinger/Vol recomputed
    df.Change_Pct = df.Close.pct_change() * 100   // match training CSV units

    IF hmm_state_df IS NOT EMPTY THEN
        df = df.merge(hmm_state_df, on = "Date", how = "left")
        df.HMM_State       = fillna(df.HMM_State, 2)         // default Neutral
        top_labels         = value_counts(df.HMM_State_Label).head(5)
        FOR label IN top_labels DO
            df["HMM_" + normalize(label)] = (df.HMM_State_Label == label) // one-hot dummies
        END FOR
        drop df.HMM_State_Label
    END IF

    IF sent_df IS NOT EMPTY THEN
        df = df.merge(sent_df, on = "Date", how = "left")
        fillna(df.[Article_Count, Positive_Prob, Negative_Prob,
                   Neutral_Prob, Sentiment_Score], 0.0)
        df.Confidence = max(Positive_Prob, Negative_Prob, Neutral_Prob)
    END IF

    df = engineer_all_features(df)            // cyclical + lags + interactions
    RETURN sort_by_date(df)
END
