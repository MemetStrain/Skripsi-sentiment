ALGORITHM run_sentiment_on_articles

ALGORITHM run_sentiment_on_articles
SOURCE: scheduler/sentiment_runner.py :: run_sentiment_on_articles

INPUT:
    articles  // list[dict] hasil preprocess (title + content)

OUTPUT:
    articles dimutasi in-place dengan field:
        sentiment_label, sentiment_score,
        positive_prob, negative_prob, neutral_prob,
        combined_confidence,
        title_sentiment, title_confidence,
        title_{positive,negative,neutral}_prob

BEGIN
    IF articles IS EMPTY THEN RETURN articles

    (model, tokenizer, device) = _load_model()       // lazy, sekali per proses

    FOR i = 0 TO |articles| - 1 STEP BATCH_SIZE DO   // BATCH_SIZE = 16
        batch = articles[i : i + BATCH_SIZE]

        flat_titles = []
        meta        = []                              // per-article: title_idx or NULL
        FOR a IN batch DO
            title = strip(a.title OR "")
            IF title IS NOT EMPTY THEN
                meta.append(|flat_titles|)
                flat_titles.append(title)
            ELSE
                meta.append(NULL)
            END IF
        END FOR

        TRY
            probs = _score_texts(flat_titles, model, tokenizer, device)
        CATCH Exception
            FOR a IN batch DO
                a.setdefault(sentiment_label, "Neutral")
                a.setdefault(sentiment_score, 0.0)
                a.setdefault(positive_prob, 0.33)
                a.setdefault(negative_prob, 0.33)
                a.setdefault(neutral_prob, 0.34)
            END FOR
            CONTINUE                                  // next batch
        END TRY

        FOR (a, title_idx) IN zip(batch, meta) DO
            IF title_idx IS NULL THEN
                combined = [1.0, 0.0, 0.0]            // degenerate Neutral
            ELSE
                combined = probs[title_idx]            // shape (3,)
            END IF

            label_idx = argmax(combined)
            label     = LABELS[label_idx]              // {0:Neutral, 1:Positive, 2:Negative}
            (neu, pos, neg) = (combined[0], combined[1], combined[2])
            score     = pos - neg

            a.update({
                sentiment_label     : label,
                sentiment_score     : round(score, 4),
                positive_prob       : round(pos, 4),
                negative_prob       : round(neg, 4),
                neutral_prob        : round(neu, 4),
                combined_confidence : round(max(combined), 4) })

            IF title_idx IS NOT NULL THEN
                a.update({
                    title_sentiment     : label,
                    title_confidence    : round(max(combined), 4),
                    title_neutral_prob  : round(neu, 4),
                    title_positive_prob : round(pos, 4),
                    title_negative_prob : round(neg, 4) })
            END IF
        END FOR
    END FOR

    RETURN articles
END


PROCEDURE _load_model
SOURCE: scheduler/sentiment_runner.py :: _load_model

BEGIN
    device    = torch.device("cuda" IF torch.cuda.is_available() ELSE "cpu")
    tokenizer = AutoTokenizer.from_pretrained(MODEL_NAME)     // yiyanghkust/finbert-tone
    model     = AutoModelForSequenceClassification.from_pretrained(MODEL_NAME)
    model.to(device)
    model.eval()
    RETURN (model, tokenizer, device)
END


PROCEDURE _score_texts
SOURCE: scheduler/sentiment_runner.py :: _score_texts

INPUT:
    texts, model, tokenizer, device

OUTPUT:
    probs  // ndarray shape (N, 3) — Neutral, Positive, Negative

BEGIN
    IF texts IS EMPTY THEN RETURN zeros((0, 3))

    inputs = tokenizer(texts,
                       return_tensors = "pt",
                       truncation     = TRUE,
                       max_length     = MAX_LENGTH,          // 512
                       padding        = TRUE)
    inputs = move_to_device(inputs, device)

    WITH torch.no_grad() DO
        outputs = model(**inputs)
        probs   = torch.softmax(outputs.logits, dim = -1).cpu().numpy()
    END WITH

    RETURN probs
END

ALGORITHM compute_sentiment_aggregates

ALGORITHM compute_sentiment_aggregates
SOURCE: scheduler/sentiment_runner.py :: compute_sentiment_aggregates

INPUT:
    articles  // list[dict] dengan field sentimen lengkap

OUTPUT:
    aggregates  // list[dict] per-tanggal (Daily)

BEGIN
    daily = defaultdict(LAMBDA {
        positive_sum : 0.0, negative_sum : 0.0, neutral_sum : 0.0,
        score_sum    : 0.0, count        : 0 })

    FOR a IN articles DO
        d = a.get(date, "")
        IF d IS EMPTY OR "sentiment_score" NOT IN a
                          OR "positive_prob"   NOT IN a THEN
            CONTINUE
        END IF
        bucket = daily[d]
        bucket.positive_sum += a.positive_prob
        bucket.negative_sum += a.negative_prob
        bucket.neutral_sum  += a.neutral_prob
        bucket.score_sum    += a.sentiment_score
        bucket.count        += 1
    END FOR

    aggregates = []
    FOR (date_str, b) IN sorted(daily.items()) DO
        n     = b.count
        pos   = b.positive_sum / n
        neg   = b.negative_sum / n
        neu   = b.neutral_sum  / n
        score = b.score_sum    / n
        dominant = argmax_by_value([
                       ("Positive", pos),
                       ("Negative", neg),
                       ("Neutral",  neu) ])

        aggregates.append({
            date                : date_str,
            frequency           : "Daily",
            article_count       : n,
            positive_prob       : round(pos,   4),
            negative_prob       : round(neg,   4),
            neutral_prob        : round(neu,   4),
            sentiment_score     : round(score, 4),
            dominant_sentiment  : dominant })
    END FOR

    RETURN aggregates
END

Catatan Implementasi

  • Title-only: body artikel tidak diskor. Combined_* mirror dari Title_*.
  • Empty-title: degenerate Neutral [1.0, 0.0, 0.0]; field title_* dilewati.
  • Label space: LABELS = ["Neutral", "Positive", "Negative"] mengikuti yiyanghkust/finbert-tone id2label (BERBEDA dari ProsusAI/finbert).
  • Batch exception fallback: tiap artikel di batch mendapat default Neutral (0.33 / 0.33 / 0.34) bila inferensi gagal — tidak crash seluruh proses.
  • Cross-reference: flowchart P3 untuk alur kontrol; structure chart P3 untuk dekomposisi modul.
