Pendahuluan

Pseudocode ini mengikuti flowchart P4. Entry-point update_hmm_states(db, write_existing=False) memuat parameter HMM beku dari D5, merekonstruksi GaussianHMM, lalu menjalankan online forward filter (bukan Viterbi) untuk mendecode state per tanggal. Helper privat (_build_hmm_features, _hmm_from_params, _forward_filter, plus indikator _wilder_rsi/_macd_line/_bb_width_absolute/_rolling_zscore) ditampilkan ringkas.

ALGORITHM update_hmm_states

ALGORITHM update_hmm_states
SOURCE: scheduler/hmm_updater.py :: update_hmm_states

INPUT:
    db                 // Firestore client
    write_existing = FALSE

OUTPUT:
    side effects on D4 (hmm_states); log lines

BEGIN
    // (1) Load frozen HMM parameters from D5
    params = read_hmm_params(db, FREQUENCY)              // FREQUENCY = "Daily"
    IF params IS NULL THEN
        log_error("hmm_models/Daily missing — run migrate_hmm_to_firestore first")
        RETURN
    END IF

    feat_cols      = params.feat_cols
    vol_window     = int(params.volatility_window)
    norm_window    = int(params.norm_window)
    state_to_label = { int(k): v FOR (k, v) IN params.state_to_label.items() }

    // (2) Pull all prices from D1
    price_docs = db.collection("daily_prices").order_by("date").stream()
    price_rows = []
    FOR doc IN price_docs DO
        d = doc.to_dict()
        price_rows.append({ date: d.date, close: float(d.close) })
    END FOR

    IF |price_rows| < 50 THEN
        log_warning("Not enough price data for HMM (" + |price_rows| + " rows)")
        RETURN
    END IF

    price_df = DataFrame(price_rows)

    // (3) Build 5 HMM features (causal Z-score)
    feat_df = _build_hmm_features(price_df, vol_window, norm_window)
    IF |feat_df| < 30 THEN
        log_warning("Skipping HMM update: too few feature rows")
        RETURN
    END IF

    X = feat_df[feat_cols].values.astype(float64)         // shape (T, 5)

    // (4) Reconstruct GaussianHMM + ONE-SHOT forward filter
    model  = _hmm_from_params(params)
    states = _forward_filter(model, X)                    // shape (T,) int

    // (5) Decide which rows to write
    existing_dates = ∅
    IF NOT write_existing THEN
        FOR doc IN db.collection("hmm_states").stream() DO
            d = doc.to_dict() OR {}
            IF d.frequency == FREQUENCY THEN
                existing_dates.add(str(d.date))
            END IF
        END FOR
    END IF

    to_write = []
    FOR (i, row) IN feat_df.iterrows() DO
        date_str = format(row.Date, "%Y-%m-%d")
        IF NOT write_existing AND date_str IN existing_dates THEN
            CONTINUE
        END IF
        s = int(states[i])
        to_write.append({
            date        : date_str,
            frequency   : FREQUENCY,
            state       : s,
            state_label : state_to_label.get(s, "Neutral"),
            log_return  : round(float(row.Log_Return_Z), 4),
            volatility  : round(float(row.Volatility_Z), 4),
            rsi         : round(float(row.RSI_norm),     4) })
    END FOR

    IF to_write IS EMPTY THEN
        log_info("HMM Daily: no new dates to write")
        RETURN
    END IF

    write_hmm_states_batch(db, to_write)                 // → D4
    log_info("HMM Daily: wrote " + |to_write| + " new state docs")
END

PROCEDURE _build_hmm_features

PROCEDURE _build_hmm_features
SOURCE: scheduler/hmm_updater.py :: _build_hmm_features

INPUT:
    price_df, vol_window, norm_window

OUTPUT:
    feat_df  // columns: Date, Log_Return_Z, Volatility_Z, RSI_norm, MACD_Z, BB_Width_Z

BEGIN
    df    = price_df.sort_values("date").reset_index(drop=TRUE).copy()
    close = df.close.astype(float)

    log_return = log(close / close.shift(1))
    volatility = log_return.rolling(vol_window, min_periods=2).std()
    rsi        = _wilder_rsi(close, period=14)
    macd       = _macd_line(close, fast=12, slow=26)
    bb_width   = _bb_width_absolute(close, window=20, n_std=2.0)

    feat = DataFrame({
        Date         : to_datetime(df.date),
        Log_Return_Z : _rolling_zscore(log_return, norm_window),
        Volatility_Z : _rolling_zscore(volatility, norm_window),
        RSI_norm     : (rsi - 50.0) / 50.0,
        MACD_Z       : _rolling_zscore(macd, norm_window),
        BB_Width_Z   : _rolling_zscore(bb_width, norm_window) })

    feat = feat.replace([inf, -inf], NaN).dropna().reset_index(drop=TRUE)
    RETURN feat
END


PROCEDURE _hmm_from_params
SOURCE: scheduler/hmm_updater.py :: _hmm_from_params

INPUT:
    params  // dict dari D5

OUTPUT:
    model  // hmm.GaussianHMM tanpa fit

BEGIN
    model = hmm.GaussianHMM(
        n_components    = int(params.n_components),
        covariance_type = params.covariance_type,
        init_params     = "",          // jangan timpa nilai seeded
        params          = "stmc",
        min_covar       = 1e-3)
    model.startprob_ = np.asarray(params.startprob_, dtype=float)
    model.transmat_  = np.asarray(params.transmat_,  dtype=float)
    model.means_     = np.asarray(params.means_,     dtype=float)
    model.covars_    = np.asarray(params.covars_,    dtype=float)
    RETURN model
END


PROCEDURE _forward_filter
SOURCE: scheduler/hmm_updater.py :: _forward_filter

INPUT:
    model, X  // X shape (T, 5)

OUTPUT:
    states  // shape (T,) int — argmax_i P(q_t = i | O_1..O_t)

BEGIN
    log_emit  = model._compute_log_likelihood(X)              // shape (T, K)
    log_start = log(max(model.startprob_, 1e-300))
    log_trans = log(max(model.transmat_,  1e-300))

    (T, K)    = log_emit.shape
    log_alpha = empty((T, K))
    log_alpha[0] = log_start + log_emit[0]
    FOR t = 1 TO T - 1 DO
        log_alpha[t] = logsumexp(log_alpha[t - 1][:, NEW] + log_trans, axis=0)
                       + log_emit[t]
    END FOR
    RETURN argmax(log_alpha, axis=1)
END

Catatan Implementasi

  • Forward filter, BUKAN Viterbi: hanya O_1..O_t dipakai untuk state pada waktu t → histori tidak berubah seiring waktu.
  • One-shot, bukan per-baris: _forward_filter dipanggil sekali atas matriks X penuh; loop per-baris sesudahnya hanya untuk filter tanggal yang sudah ada di D4 dan membangun to_write[].
  • Guard data-cukup: |price_rows| < 50 dan |feat_df| < 30 → early-return log.warning.
  • Idempotensi: default write_existing=False → tanggal yang sudah ada di D4 di-skip; rerun aman tanpa duplikasi.
