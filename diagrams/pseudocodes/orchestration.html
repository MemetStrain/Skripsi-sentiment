Pendahuluan

Pseudocode ini mengikuti flowchart Orchestration — alur kontrol untuk --mode daily di scheduler/main.py::run_daily_update(db). Tiga sub-prosedur ditampilkan eksplisit: _step_price (P1), _step_news (P2 + memanggil P3 run_sentiment_on_articles), reconcile CSV → Firestore (Langkah 3, single source of truth), pemicu P4 (update_hmm_states), dan langkah baru P5 (_step_precompute → precompute_and_write → inference.compute_forecast_trails) yang dibungkus try/except. Detail internal P3, P4, dan P5 berada di berkas pseudocode masing-masing.

ALGORITHM main · scheduler/main.py

ALGORITHM main
SOURCE: scheduler/main.py :: main

INPUT:
    sys.argv  // --mode {initial, daily, rebuild-hmm}, --reset-progress

OUTPUT:
    side effects (CSV writes, Firestore writes, log); exit code

BEGIN
    args = parse_argparse()
    IF args.reset_progress THEN
        _reset_progress()
        RETURN
    END IF

    IF args.mode == "rebuild-hmm" THEN
        migrate_hmm_to_firestore.main()        // republish frozen params
        RETURN
    END IF

    TRY
        init_firebase()                        // env JSON / file / ADC fallback
    CATCH Exception
        log_error("Firebase init failed")
        sys.exit(1)
    END TRY

    db = firestore.client()

    IF args.mode == "initial" THEN
        run_initial_load(db)                   // bulk CSV → Firestore (4 checkpointed steps)
    ELSE
        run_daily_update(db)                   // the main daily pipeline (see below)
    END IF
END

ALGORITHM run_daily_update · scheduler/main.py

ALGORITHM run_daily_update
SOURCE: scheduler/main.py :: run_daily_update

INPUT:
    db   // Firestore client

OUTPUT:
    side effects on D1 (daily_prices), D2 (news_articles),
    D3 (sentiment_aggregates), D4 (hmm_states) + 3 local CSV files

BEGIN
    paths = _paths()      // {cpo_raw, cpo_preproc, news_raw, news_preproc, news_sent}

    // ── Step 1 · P1: PRICE ─────────────────────────────────────
    price_changed = _step_price(paths)                        // CSV-only; tidak menulis Firestore

    // ── Step 2 · P2: NEWS (scrape + preprocess + sentiment) ────
    (_, news_changed) = _step_news(paths)                     // append 3 CSV; tidak menulis Firestore

    // ── Step 3 · Reconcile CSV → Firestore (single source of truth) ─
    // Diff full CSV vs Firestore untuk prices + news + sentiment_aggregates;
    // tulis hanya dokumen yang missing/stale. Self-healing terhadap skip-run.
    recon_writes = reconcile_all(db, paths)

    // ── Step 4 · P4: HMM state decoding ────────────────────────
    update_hmm_states(db)                                     // see P4 pseudocode

    // ── Step 5 · P5: Forecast precompute (try/except, isolated) ─
    // Skip jika tidak ada perubahan di langkah 1–3; deterministic doc IDs
    // menjamin re-run idempotent.
    IF price_changed OR news_changed OR recon_writes > 0 THEN
        _step_precompute(db)                                  // see Precompute pseudocode
    ELSE
        log_info("no data changes; forecasts already current — skip")
    END IF

    RETURN log_DAILY_UPDATE_COMPLETE
END


PROCEDURE _step_precompute
SOURCE: scheduler/main.py :: _step_precompute

INPUT:
    db   // Firestore client

OUTPUT:
    bool — TRUE jika ≥1 dokumen forecast point ditulis

BEGIN
    TRY
        n = precompute_forecasts.precompute_and_write(db)     // see P5 pseudocode
    CATCH Exception AS e
        log_exception("forecast precompute failed: " + str(e))
        RETURN FALSE                                        // jangan rollback langkah 1–4
    END TRY
    RETURN n > 0
END

PROCEDURE _step_price · P1 (linear)

PROCEDURE _step_price
SOURCE: scheduler/main.py :: _step_price

INPUT:
    db, paths

OUTPUT:
    bool — TRUE jika harga baru ditulis, FALSE bila skip

BEGIN
    cutoff = most_recent_trading_day()                       // weekday-only floor
    have   = latest_price_date(paths.cpo_raw)

    IF have IS NOT NULL AND have >= cutoff THEN
        RETURN FALSE                                       // skip-if-current
    END IF

    price = fetch_latest_price()                             // investiny, lookback 90/365/1095 d
    IF price IS NULL THEN RETURN FALSE
    IF have IS NOT NULL AND price.date <= have THEN
        RETURN FALSE                                       // stale fetch — already have it
    END IF

    appended = append_price_row_indonesian(paths.cpo_raw, price)
    IF NOT appended THEN RETURN FALSE             // duplicate date guard

    preprocess_price_csv(paths.cpo_raw, paths.cpo_preproc)    // build engineered CSV
    write_price(db, price)                                    // daily_prices/{date} ← D1
    RETURN TRUE
END

PROCEDURE _step_news · P2 (linear, calls P3a)

PROCEDURE _step_news
SOURCE: scheduler/main.py :: _step_news

INPUT:
    paths

OUTPUT:
    (scored: list[dict], news_changed: bool)

BEGIN
    cutoff = most_recent_trading_day()
    have   = latest_news_date(paths.news_sent)

    IF have IS NOT NULL AND have >= cutoff THEN
        RETURN ([], FALSE)
    END IF

    raw = scrape_new_articles(have)                          // MPOB, multi-keyword × page, dedup URL
    IF raw IS EMPTY THEN RETURN ([], FALSE)

    raw_rows = [ article_to_raw_row(a) FOR a IN raw ]
    append_news_rows(paths.news_raw, raw_rows, RAW_FIELDS)   // arsip mentah

    cleaned = preprocess_articles(raw)                       // NewsPreprocessor, drops empty
    IF cleaned IS EMPTY THEN RETURN ([], FALSE)

    cleaned_rows = [ article_to_raw_row(a) FOR a IN cleaned ]
    append_news_rows(paths.news_preproc, cleaned_rows, RAW_FIELDS)

    scored = run_sentiment_on_articles(cleaned)              // P3a — see P3 pseudocode
    sent_rows = [ article_to_sentiment_row(a) FOR a IN scored ]
    append_news_rows(paths.news_sent, sent_rows, SENTIMENT_FIELDS)

    RETURN (scored, TRUE)
END
